import 'package:network_info_plus/network_info_plus.dart';

class WiFiQRConfig {
  String? ssid;
  String? bssid;  // MAC address cá»§a router
  
  Future<bool> getWiFiInfo() async {
    // Check if connected to WiFi
    var connectivity = await Connectivity().checkConnectivity();
    if (connectivity != ConnectivityResult.wifi) {
      print("âŒ Not connected to WiFi");
      return false;
    }
    
    final info = NetworkInfo();
    
    // Get WiFi name (SSID)
    String? wifiName = await info.getWifiName();
    if (wifiName != null) {
      // Remove quotes if present
      ssid = wifiName.replaceAll('"', '');
    }
    
    // Get router MAC address (BSSID)
    String? wifiBSSID = await info.getWifiBSSID();
    if (wifiBSSID != null) {
      // Remove colons: "e0:d4:62:e6:63:b0" â†’ "e0d462e663b0"
      bssid = wifiBSSID.replaceAll(':', '');
    }
    
    if (ssid != null && bssid != null) {
      print("âœ… WiFi Info:");
      print("  SSID: $ssid");
      print("  BSSID: $bssid");
      return true;
    }
    
    return false;
  }
  
  String generateQRData({required String password}) {
    return json.encode({
      "BS": bssid,           // Router MAC
      "P": password,         // WiFi password
      "U": "15463733-OEM",   // User ID
      "RS": ssid             // WiFi name
    });
  }
  
  Widget buildQRCodeWidget({
    required String password,
    double size = 300,
  }) {
    String qrData = generateQRData(password: password);
    
    return QrImageView(
      data: qrData,
      version: QrVersions.auto,
      size: size,
      backgroundColor: Colors.white,
    );
  }
}
